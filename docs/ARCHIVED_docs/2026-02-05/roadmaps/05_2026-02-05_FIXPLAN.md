#### Step 1: Implémenter les calculs de métriques

Fichier: [backtests/metrics.py](backtests/metrics.py)

```python
import numpy as np
import pandas as pd
from dataclasses import dataclass

@dataclass
class BacktestMetrics:
    start_date: str
    end_date: str
    total_return: float
    sharpe_ratio: float
    max_drawdown: float
    win_rate: float
    profit_factor: float
    total_trades: int
    avg_trade_duration: float = None
    calmar_ratio: float = None
    sortino_ratio: float = None
    
    @classmethod
    def from_returns(cls, returns: pd.Series, trades: list, start_date: str, end_date: str) -> 'BacktestMetrics':
        """
        Calculer les métriques à partir des returns.
        
        Args:
            returns: Series de returns quotidiens (0.01 = 1%)
            trades: List of trade P&L values
            start_date, end_date: Period
        
        Returns:
            BacktestMetrics avec tous les champs remplis
        """
        # Total return
        total_return = (1 + returns).prod() - 1
        
        # Sharpe ratio (252 trading days/year)
        sharpe_ratio = returns.mean() / returns.std() * np.sqrt(252)
        
        # Max drawdown
        cumulative = (1 + returns).cumprod()
        running_max = cumulative.expanding().max()
        drawdown = (cumulative - running_max) / running_max
        max_drawdown = drawdown.min()
        
        # Win rate
        winning_trades = sum(1 for pnl in trades if pnl > 0)
        win_rate = winning_trades / len(trades) if trades else 0.0
        
        # Profit factor
        gross_profit = sum(pnl for pnl in trades if pnl > 0)
        gross_loss = abs(sum(pnl for pnl in trades if pnl < 0))
        profit_factor = gross_profit / gross_loss if gross_loss > 0 else 0.0
        
        # Sortino ratio (apenas downside volatility)
        downside_returns = returns[returns < 0]
        sortino_ratio = returns.mean() / downside_returns.std() * np.sqrt(252) if len(downside_returns) > 0 else 0.0
        
        # Calmar ratio (return / max drawdown)
        calmar_ratio = total_return / abs(max_drawdown) if max_drawdown < 0 else 0.0
        
        return cls(
            start_date=start_date,
            end_date=end_date,
            total_return=total_return,
            sharpe_ratio=sharpe_ratio,
            max_drawdown=max_drawdown,
            win_rate=win_rate,
            profit_factor=profit_factor,
            total_trades=len(trades),
            avg_trade_duration=None,  # TODO
            calmar_ratio=calmar_ratio,
            sortino_ratio=sortino_ratio
        )
    
    def summary(self) -> str:
        """Return formatted metrics summary."""
        return f"""
╔════════════════════════════════════════════╗
║         BACKTEST METRICS SUMMARY            ║
╠════════════════════════════════════════════╣
║ Period: {self.start_date} to {self.end_date}
║ Total Return:       {self.total_return:>7.2%}
║ Sharpe Ratio:       {self.sharpe_ratio:>7.2f}
║ Sortino Ratio:      {self.sortino_ratio:>7.2f}
║ Max Drawdown:       {self.max_drawdown:>7.2%}
║ Calmar Ratio:       {self.calmar_ratio:>7.2f}
║
║ Win Rate:           {self.win_rate:>7.2%}
║ Profit Factor:      {self.profit_factor:>7.2f}
║ Total Trades:       {self.total_trades:>7d}
╚════════════════════════════════════════════╝
        """
```

```python
import pandas as pd
import numpy as np
from backtests.metrics import BacktestMetrics

class BacktestRunner:
    def run(self, symbols: list, start_date: str = None, end_date: str = None) -> BacktestMetrics:
        """Run full backtest workflow."""
        
        if start_date is None:
            start_date = self.config.start_date
        if end_date is None:
            end_date = self.config.end_date
        
        logger.info("backtest_starting", symbols=symbols, start_date=start_date, end_date=end_date)
        
        # 1. Load data
        price_data = {}
        for symbol in symbols:
            try:
                df = self.loader.load_IBKR API_data(
                    broker_name='IBKR',
                    symbol=symbol,
                    timeframe='1d'
                )
                price_data[symbol] = df['close']
            except Exception as e:
                logger.error("data_load_failed", symbol=symbol, error=str(e))
                continue
        
        if not price_data:
            raise ValueError("No data loaded")
        
        # 2. Align dates
        prices_df = pd.DataFrame(price_data)
        prices_df = prices_df[(prices_df.index >= start_date) & (prices_df.index <= end_date)]
        
        # 3. Generate signals
        signals = self.strategy.generate_signals(prices_df)
        logger.info("backtest_signals_generated", count=len(signals))
        
        # 4. Execute backtest logic
        trades = []  # List of {entry_price, exit_price, pnl, duration}
        portfolio_value = [self.config.initial_capital]
        daily_returns = []
        
        for index, row in prices_df.iterrows():
            # Mark to market
            self.risk_engine.mark_to_market({sym: row[sym] for sym in symbols})
            
            # Process signals
            for signal in signals:
                if signal.side == "entry":
                    position_size = self._calculate_position_size(portfolio_value[-1], signal)
                    self.risk_engine.register_entry(
                        signal.symbol_pair,
                        entry_price=row[signal.symbol_pair],
                        quantity=position_size,
                        side=signal.side
                    )
                elif signal.side == "exit":
                    # Close position
                    position = self.risk_engine.positions.get(signal.symbol_pair)
                    if position:
                        exit_price = row[signal.symbol_pair]
                        pnl = position.quantity * (exit_price - position.entry_price)
                        trades.append({'pnl': pnl})
                        self.risk_engine.register_exit(signal.symbol_pair, exit_price, pnl)
            
            # Calculate daily P&L
            total_pnl = sum(pos.pnl for pos in self.risk_engine.positions.values())
            daily_value = portfolio_value[-1] + total_pnl
            portfolio_value.append(daily_value)
            daily_returns.append((daily_value - portfolio_value[-2]) / portfolio_value[-2])
        
        # 5. Calculate metrics
        returns = pd.Series(daily_returns)
        metrics = BacktestMetrics.from_returns(
            returns=returns,
            trades=[t['pnl'] for t in trades],
            start_date=start_date,
            end_date=end_date
        )
        
        logger.info("backtest_completed", metrics=metrics.__dict__)
        return metrics
    
    def _calculate_position_size(self, equity: float, signal) -> float:
        """Calculate position size based on volatility and risk."""
        # Simple volatility-adjusted sizing
        risk_amount = equity * self.config.max_risk_per_trade
        volatility = 0.02  # TODO: Calculate from data
        position_size = risk_amount / volatility
        return position_size
```

```python
    """Paper trading in sandbox mode."""
    logger.info("paper_trading_mode_starting")
    
    from data.loader import DataLoader
    from strategies.pair_trading import PairTradingStrategy
    from risk.engine import RiskEngine
    from execution.IBKR API_engine import IBKR APIExecutionEngine

                sandbox=settings.execution.use_sandbox,
                engine=settings.execution.engine,
                symbols=symbols)
    
    # Live loop
    try:
        while True:
            # Load latest data
            prices = {}
            for symbol in symbols:
                try:
                    df = loader.load_IBKR API_data(

                    )
                    prices[symbol] = df['close']

                    logger.error("data_load_failed", symbol=symbol, error=str(e))

                continue

                )

                    continue

                current_price = prices_df[signal.symbol_pair].iloc[-1]

                    order_id=f"paper_{datetime.now().timestamp()}",

                except Exception as e:
                    logger.error("order_submission_failed", error=str(e))
```

### 5.2 Solution: Parallélisation & Vectorisation

#### Step 1: Paralléliser Pair Discovery

Fichier: [strategies/pair_trading.py](strategies/pair_trading.py)

```python
from multiprocessing import Pool, cpu_count
import functools

class PairTradingStrategy(BaseStrategy):
    
    def find_cointegrated_pairs_parallel(
        self,
        price_data: pd.DataFrame,
        lookback: int = None,
        num_workers: int = None
    ) -> List[Tuple[str, str, float]]:
        """
        Find cointegrated pairs using multiprocessing.
        5-10x faster than sequential version.
        """
        if lookback is None:
            lookback = self.config.lookback_window
        
        if num_workers is None:
            num_workers = cpu_count() - 1  # Leave 1 core free
        
        data = price_data.tail(lookback)
        symbols = data.columns.tolist()
        
        # Generate all pairs
        pairs_to_test = []
        for i, sym1 in enumerate(symbols):
            for j, sym2 in enumerate(symbols[i+1:], start=i+1):
                pairs_to_test.append((sym1, sym2, data[sym1], data[sym2]))
        
        logger.info("pair_discovery_starting", 
                   total_pairs=len(pairs_to_test),
                   workers=num_workers)
        
        # Test pairs in parallel
        with Pool(num_workers) as pool:
            results = pool.map(self._test_pair_cointegration, pairs_to_test)
        
        # Filter results
        cointegrated_pairs = [r for r in results if r is not None]
        
        logger.info("pair_discovery_complete",
                   cointegrated_count=len(cointegrated_pairs))
        
        return cointegrated_pairs
    
    @staticmethod
    def _test_pair_cointegration(pair_data):
        """Test single pair (runs in worker process)."""
        sym1, sym2, series1, series2 = pair_data
        
        try:
            corr = series1.corr(series2)
            
            if abs(corr) < 0.7:  # Min correlation threshold
                return None
            
            result = engle_granger_test(series1, series2)
            
            if result['is_cointegrated']:
                hl = half_life_mean_reversion(pd.Series(result['residuals']))
                
                if hl and hl <= 60:  # Max half-life
                    return (sym1, sym2, result['adf_pvalue'], hl)
        
        except Exception:
            pass
        
        return None
```

#### Step 2: Vectoriser Signal Generation

```python
def generate_signals_vectorized(self, market_data: pd.DataFrame) -> List[Signal]:
    """
    Generate signals using vectorized pandas operations.
    2-3x faster than looping.
    """
    signals = []
    
    # Découvrir paires (now parallelized)
    cointegrated = self.find_cointegrated_pairs_parallel(market_data)
    
    # Vectorized spread computation
    for sym1, sym2, pvalue, hl in cointegrated:
        pair_key = f"{sym1}_{sym2}"
        
        try:
            y = market_data[sym1]
            x = market_data[sym2]
            
            # Vectorized spread calculation
            model = SpreadModel(y, x)
            self.spread_models[pair_key] = model
            
            spread = model.compute_spread(y, x)
            z_score = model.compute_z_score(spread, lookback=20)
            
            # Vectorized signal generation
            current_z = z_score.iloc[-1]
            
            # Entry signals
            if current_z > self.config.entry_z_score and pair_key not in self.active_trades:
                signals.append(Signal(
                    symbol_pair=pair_key,
                    side="short",
                    strength=min(abs(current_z) / 3.0, 1.0),
                    reason=f"Z={current_z:.2f}"
                ))
            
            elif current_z < -self.config.entry_z_score and pair_key not in self.active_trades:
                signals.append(Signal(
                    symbol_pair=pair_key,
                    side="long",
                    strength=min(abs(current_z) / 3.0, 1.0),
                    reason=f"Z={current_z:.2f}"
                ))
            
            # Exit signals
            if pair_key in self.active_trades:
                if abs(current_z) <= self.config.exit_z_score:
                    signals.append(Signal(
                        symbol_pair=pair_key,
                        side="exit",
                        strength=1.0,
                        reason=f"Mean reversion Z={current_z:.2f}"
                    ))
        
        except Exception as e:
            logger.error("signal_generation_failed", pair=pair_key, error=str(e))
    
    return signals
```

```python
import pickle
from pathlib import Path
from datetime import datetime, timedelta

class PairTradingStrategy(BaseStrategy):
    
    def __init__(self):
        super().__init__()
        self.cache_dir = Path("cache/pairs")
        self.cache_dir.mkdir(parents=True, exist_ok=True)
    
    def load_cached_pairs(self, max_age_hours=24):
        """Load cached cointegrated pairs if recent."""
        cache_file = self.cache_dir / "cointegrated_pairs.pkl"
        
        if cache_file.exists():
            mod_time = datetime.fromtimestamp(cache_file.stat().st_mtime)
            age = datetime.now() - mod_time
            
            if age < timedelta(hours=max_age_hours):
                with open(cache_file, 'rb') as f:
                    pairs = pickle.load(f)
                logger.info("loaded_cached_pairs", pairs=len(pairs), age_hours=age.total_seconds()/3600)
                return pairs
        
        return None
    
    def find_cointegrated_pairs(self, price_data, lookback=None):
        """Find pairs with caching."""
        
        # Try cache first
        cached = self.load_cached_pairs()
        if cached is not None:
            return cached
        
        # Otherwise discover and cache
        pairs = self.find_cointegrated_pairs_parallel(price_data, lookback)
        
        # Cache result
        cache_file = self.cache_dir / "cointegrated_pairs.pkl"
        with open(cache_file, 'wb') as f:
            pickle.dump(pairs, f)
        
        return pairs
```
